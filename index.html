function fibonacci(n) {
  if (n < 0) return "Número inválido";
  if (n === 0) return 0;
  if (n === 1) return 1;
                    
  let a = 0,
    b = 1,
    temp;
    for (let i = 2; i <= n; i++) {
    temp = a + b;
    a = b;
    b = temp;
    }
    return b;
}

        num1 = Number(num1);
        num2 = Number(num2);
        num3 = Number(num3);
                          
        let numeros = [num1, num2, num3];
        numeros.sort((a, b) => b - a);

    }

function factorial(n) {
    if (n < 0) return "Número inválido";
    let resultado = 1;
    for (let i = 1; i <= n; i++) {
        resultado *= i;
    }
    return resultado;
    }

function esPrimo(num) {
  if (num < 2) return false;
  for (let i = 2; i <= Math.sqrt(num); i++) {
    if (num % i === 0) {
      return false;
    }
  }
  return true;
}
              
function verificarPrimo() {
  let num = parseInt(document.getElem("numerop").value);
  let resultadoTexto = document.getElem("resultadop");
              
  if (isNaN(num) || num <= 0) {
    resultadoTexto.textContent =
      "Por favor, ingresa un número entero positivo";
    return;
  }
              
  if (esPrimo(num)) {
    resultadoTexto.textContent = num + " es un primo";
  } else {
    resultadoTexto.textContent = num + " nonúmero primo";
  }
}
