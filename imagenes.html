function fibonacci(n) {
  if (n < 0) return "Número inválido";
  if (n === 0) return 0;
  if (n === 1) return 1;
                    
  let a = 0,
    b = 1,
    temp;
    for (let i = 2; i <= n; i++) {
    temp = a + b;
    a = b;
    b = temp;
    }
    return b;
}
                    
function calcularFibonacci() {
    let numero = parseInt(document.getElemen("numero").value);
    let resultado = fibonacci(numero);
    document.getElementById("resultatextContent = resultado;
}
